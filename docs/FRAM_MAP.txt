Szczegóły sekcji
=============================================================

HEADER (0x0000, 32B)
  uint32  magic           "DOZA" = 0x444F5A41
  uint16  layout_version  8
  uint16  channel_count
  uint32  init_timestamp
  uint32  last_write
  uint8   flags
  uint8   _reserved[11]
  uint32  header_crc

ACTIVE_CONFIG / PENDING_CONFIG — ChannelConfig (32B × 8)
  uint32  events_bitmask  bity 2,4,6,...,22 = godziny zdarzeń
  uint8   days_bitmask    bity 0-6 = Pon-Ndz
  uint8   _reserved1[3]
  float   daily_dose_ml
  float   dosing_rate     ml/s
  uint8   enabled
  uint8   has_pending
  uint8   _reserved2[2]
  uint32  crc32
  uint8   _padding[8]

DAILY_STATE — ChannelDailyState (24B × 8)
  uint32  events_completed  bitmask
  uint32  events_failed     bitmask
  float   today_added_ml
  uint8   last_reset_day
  uint8   failed_count
  uint8   _reserved[2]
  uint32  crc32
  uint8   _padding[4]

CONTAINER_VOLUME — ContainerVolume (8B × 8)
  uint16  container_ml    × 10 (0.1 ml precision)
  uint16  remaining_ml    × 10
  uint32  crc32

DOSED_TRACKER — DosedTracker (8B × 8)
  uint16  total_dosed_ml  × 10
  uint16  _reserved
  uint32  crc32

CHANNEL_LABELS — ChannelLabel (32B × 8)
  char    name[20]
  uint8   _reserved[8]
  uint32  crc32

CHANNEL_PARAMS — ChannelParams (32B × 8)
  float   min_single_dose_ml
  float   max_single_dose_ml
  float   max_daily_dose_ml
  float   _reserved_f
  uint8   _reserved[12]
  uint32  crc32

SHARED_NOTES (0x0B00, 400B)
  NoteEntry notes[12]       12 × 32B = 384B
    char    text[30]          treść, null-terminated
    uint8   flags             bit0: slot wypełniony
    uint8   _pad
  uint8   ch_note_idx[8]    indeks aktywnej notatki per kanał
  uint8   _pad[4]
  uint32  crc32             obejmuje wszystkie pola przed sobą (396B)

=============================================================
Notatki implementacyjne
=============================================================

- CRC32 liczy się zawsze nad sizeof(Struct) - sizeof(uint32_t) bajtami
  (uint32_t crc32 jest OSTATNIM polem każdej struktury)
- Niezgodność header.layout_version → automatyczny factory reset
- CREDENTIALS (0x0020-0x041F) kompatybilne z projektem DOLEWKA —
  ten zakres adresów nigdy nie może zostać zmieniony
- SHARED_NOTES: zapis całego bloku 400B przy każdej edycji notatki
  (FRAM MB85RC256V: 10^12 cykli zapisu — zużycie nie jest problemem)
